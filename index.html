// ============================================= // Space Wallet API - Google Apps Script // รองรับ CORS (doOptions + header ในทุก response) // ฟังก์ชัน setup() สร้างชีทและข้อมูลเริ่มต้น // ============================================= // ---------- ฟังก์ชันช่วยสร้าง JSON response พร้อม CORS ---------- function responseJson(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON) .setHeader('Access-Control-Allow-Origin', '*'); } // ---------- รองรับ Preflight Request (OPTIONS) ---------- function doOptions() { return ContentService.createTextOutput('') .setMimeType(ContentService.MimeType.TEXT) .setHeader('Access-Control-Allow-Origin', '*') .setHeader('Access-Control-Allow-Methods', 'POST, OPTIONS') .setHeader('Access-Control-Allow-Headers', 'Content-Type'); } // ---------- ทดสอบ GET ---------- function doGet() { return ContentService.createTextOutput("🌟 Space Wallet API Active") .setHeader('Access-Control-Allow-Origin', '*'); } // ---------- API หลัก (POST) ---------- function doPost(e) { var data = JSON.parse(e.postData.contents); var action = data.action; var sheet; var result = {}; try { if (action === 'getBalance') { var funds = ['ค่าเทอมลูก', 'เงินคงเหลือ']; var balances = {}; funds.forEach(function(fund) { var sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(fund); if (!sh) { balances[fund] = 0; return; } var rows = sh.getDataRange().getValues(); var bal = 0; for (var i = 1; i < rows.length; i++) { var amt = Number(rows[i][2]); if (rows[i][3] === 'รายรับ') bal += amt; else if (rows[i][3] === 'รายจ่าย') bal -= amt; } balances[fund] = bal; }); result = { success: true, balances: balances }; } else if (action === 'getTransactions') { var fund = data.fund; sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(fund); if (!sheet) return responseJson({ success: false, error: 'Sheet not found' }); var rows = sheet.getDataRange().getValues(); var transactions = []; for (var i = rows.length - 1; i >= 1; i--) { transactions.push({ row: i + 1, date: rows[i][0], description: rows[i][1], amount: rows[i][2], type: rows[i][3] }); if (data.limit && transactions.length >= data.limit) break; } var bal = 0; for (var i = 1; i < rows.length; i++) { var amt = Number(rows[i][2]); if (rows[i][3] === 'รายรับ') bal += amt; else if (rows[i][3] === 'รายจ่าย') bal -= amt; } result = { success: true, transactions: transactions, balance: bal }; } else if (action === 'addTransaction') { var fund = data.fund; sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(fund); if (!sheet) return responseJson({ success: false }); sheet.appendRow([data.date, data.description, Number(data.amount), data.type]); var rows = sheet.getDataRange().getValues(); var bal = 0; for (var i = 1; i < rows.length; i++) { var amt = Number(rows[i][2]); if (rows[i][3] === 'รายรับ') bal += amt; else if (rows[i][3] === 'รายจ่าย') bal -= amt; } result = { success: true, balance: bal }; } else if (action === 'deleteTransaction') { var fund = data.fund; sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(fund); if (!sheet) return responseJson({ success: false }); sheet.deleteRow(data.row); var rows = sheet.getDataRange().getValues(); var bal = 0; for (var i = 1; i < rows.length; i++) { var amt = Number(rows[i][2]); if (rows[i][3] === 'รายรับ') bal += amt; else if (rows[i][3] === 'รายจ่าย') bal -= amt; } result = { success: true, balance: bal }; } else { result = { success: false, error: 'Invalid action' }; } } catch (error) { result = { success: false, error: error.toString() }; } return responseJson(result); } // ---------- ฟังก์ชันสร้างชีทและข้อมูลเริ่มต้น ---------- function setup() { var ss = SpreadsheetApp.getActiveSpreadsheet(); // สร้างชีท "ค่าเทอมลูก" ถ้ายังไม่มี var sheet1 = ss.getSheetByName('ค่าเทอมลูก'); if (!sheet1) { sheet1 = ss.insertSheet('ค่าเทอมลูก'); sheet1.appendRow(['วันที่', 'รายการ', 'จำนวนเงิน', 'ประเภท']); // หัวคอลัมน์ } // สร้างชีท "เงินคงเหลือ" ถ้ายังไม่มี var sheet2 = ss.getSheetByName('เงินคงเหลือ'); if (!sheet2) { sheet2 = ss.insertSheet('เงินคงเหลือ'); sheet2.appendRow(['วันที่', 'รายการ', 'จำนวนเงิน', 'ประเภท']); } // เพิ่มข้อมูลเริ่มต้นเฉพาะถ้าชีทยังไม่มีข้อมูล (นับแถว <= 1) if (sheet1.getLastRow() <= 1) { sheet1.appendRow(['2026-07-10', 'กค69', 5500, 'รายจ่าย']); sheet1.appendRow(['2026-07-12', 'เบิกเพิ่ม', 4000, 'รายจ่าย']); } if (sheet2.getLastRow() <= 1) { sheet2.appendRow(['2026-07-09', 'ยอดยกมา', 12719, 'รายรับ']); sheet2.appendRow(['2026-07-10', 'ค่าวิทยากร นาแก', 5500, 'รายรับ']); sheet2.appendRow(['2026-07-12', 'ค่าวิทยากร เทพสถิต', 5000, 'รายรับ']); } SpreadsheetApp.getUi().alert('✅ Setup สำเร็จ!\n\nค่าเทอมลูก = -9,500 บาท (รายจ่าย)\nเงินคงเหลือ = 23,219 บาท (รายรับ)'); }
